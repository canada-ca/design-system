/* hide the check */
.can-ca-form .form-checkbox input[type="checkbox"] + label::after,
.can-ca-form .form-radio input[type="radio"] + label::after,
.can-ca-form .checkbox input[type="checkbox"] + label::after,
.can-ca-form .radio input[type="radio"] + label::after {
content: none;
}

/* unhide check when checked */
.can-ca-form .form-checkbox input[type="checkbox"]:checked + label::after,
.can-ca-form .form-radio input[type="radio"]:checked + label::after,
.can-ca-form .checkbox input[type="checkbox"]:checked + label::after,
.can-ca-form .radio input[type="radio"]:checked + label::after {
content: "";
}

/* add focus */
.can-ca-form .checkbox input[type="checkbox"]:focus + label::before,
.can-ca-form .form-checkbox input[type="checkbox"]:focus + label::before,
.can-ca-form .form-radio input[type="radio"]:focus + label::before,
.can-ca-form .radio input[type="radio"]:focus + label::before  {
box-shadow: 0 0 3pt 2pt rgb(59, 153, 252);
}


/* for green checks */

.can-ca-form .checkbox-green label::after {
content: "";
border: solid white;
border-width: 0 5px 5px 0;
}

/* for green checks */
.can-ca-form .checkbox-green input[type="checkbox"]:checked + label:before {
background-color: #318000;
}

@media only screen and (max-width: 1200px) {

.can-ca-form .form-checkbox,
.can-ca-form .form-radio {
min-width: 50%;
}

}
